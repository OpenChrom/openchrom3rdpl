<plugin>
	<groupId>org.bsc.maven</groupId>
	<artifactId>maven-processor-plugin</artifactId>
	<version>3.3.3</version>
	<executions>
		<execution>
			<id>process</id>
			<goals>
				<goal>process</goal>
			</goals>
			<phase>generate-sources</phase>
			<configuration>
				<outputDirectory>${project.basedir}/metamodel</outputDirectory>
				<sourceDirectory>${project.basedir}/src</sourceDirectory>
				<compilerArguments>-Aeclipselink.persistencexml=${project.basedir}/META-INF/persistence.xml</compilerArguments>
				<processors>
					<processor>org.eclipse.persistence.internal.jpa.modelgen.CanonicalModelProcessor</processor>
				</processors>
				<source>1.8</source>
				<target>1.8</target>
			</configuration>
		</execution>
	</executions>
	<dependencies>
		<dependency>
			<groupId>org.eclipse.persistence</groupId>
			<artifactId>org.eclipse.persistence.jpa.modelgen.processor</artifactId>
			<version>2.7.5</version>
			<scope>compile</scope>
		</dependency>
	</dependencies>
</plugin>
